// DEJAR VACÍO: el script creará el Spreadsheet automáticamente // o poner el ID de un Spreadsheet existente const HOJAS = { FICHAJES: 'Fichajes', VACACIONES: 'Vacaciones', WORKERS: 'Trabajadores', CONFIG: 'Config' }; // ════════════════════════════════════════ // ENTRYPOINTS // ════════════════════════════════════════ function doPost(e) { try { const raw = e.postData?.contents || e.parameter?.data || '{}'; const data = JSON.parse(raw); const tipo = data.tipo; // Sanitize strings to avoid cell limit (50000 chars max per cell, we cap at 500) function trunc(v, max) { if(typeof v === 'string' && v.length > (max||500)) return v.substring(0, max||500); return v; } if(data.nombre) data.nombre = trunc(data.nombre, 100); if(data.email) data.email = trunc(data.email, 200); if(data.motivo) data.motivo = trunc(data.motivo, 200); if(data.color) data.color = trunc(data.color, 50); if(data.bg) data.bg = trunc(data.bg, 80); let result; if (tipo === 'entrada') result = registrarEntrada(data); else if (tipo === 'salida') result = registrarSalida(data); else if (tipo === 'vac_request') result = guardarVacacion(data); else if (tipo === 'vac_update') result = actualizarVacacion(data); else if (tipo === 'worker_save') result = guardarWorker(data); else if (tipo === 'worker_delete') result = eliminarWorker(data); else if (tipo === 'sync_all') result = sincronizarTodo(data); else result = { ok: false, error: 'Tipo desconocido: ' + tipo }; return jsonResponse(result); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } function doGet(e) { try { const accion = e.parameter.accion; let result; if (accion === 'fichajes') result = getFichajes(e.parameter); else if (accion === 'vacaciones') result = getVacaciones(e.parameter); else if (accion === 'workers') result = getWorkers(); else if (accion === 'export') result = getExport(e.parameter); else if (accion === 'init') result = initSheet(); else result = { ok: false, error: 'Acción desconocida' }; return jsonResponse(result); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } // ════════════════════════════════════════ // SPREADSHEET HELPER // ════════════════════════════════════════ let _ss = null; function getSS() { if (_ss) return _ss; if (SS_ID) { _ss = SpreadsheetApp.openById(SS_ID); } else { // Buscar en Properties si ya creamos uno const props = PropertiesService.getScriptProperties(); const savedId = props.getProperty('SS_ID'); if (savedId) { try { _ss = SpreadsheetApp.openById(savedId); return _ss; } catch(e) {} } // Crear nuevo Spreadsheet _ss = SpreadsheetApp.create('FichaJA — Registro de Jornada'); props.setProperty('SS_ID', _ss.getId()); Logger.log('Spreadsheet creado: ' + _ss.getId()); } return _ss; } function getSheet(nombre) { const ss = getSS(); let hoja = ss.getSheetByName(nombre); if (!hoja) { hoja = ss.insertSheet(nombre); initHoja(hoja, nombre); } return hoja; } function initHoja(hoja, nombre) { const headers = { [HOJAS.FICHAJES]: ['ID','USER_ID','NOMBRE','FECHA','ENTRADA','SALIDA','HORAS_TOTAL','HORAS_ORD','HORAS_EXTRA','PAUSAS','TIMESTAMP'], [HOJAS.VACACIONES]: ['ID','USER_ID','NOMBRE','INICIO','FIN','MOTIVO','ESTADO','BY_ADMIN','TIMESTAMP'], [HOJAS.WORKERS]: ['ID','NOMBRE','EMAIL','JORNADA','COLOR','BG','ROLE','ACTIVO','PIN','TIMESTAMP'], [HOJAS.CONFIG]: ['CLAVE','VALOR','UPDATED'], }; const h = headers[nombre]; if (h) { hoja.getRange(1, 1, 1, h.length).setValues([h]); hoja.getRange(1, 1, 1, h.length) .setBackground('#1a1a1a').setFontColor('#c6f135') .setFontWeight('bold').setFontFamily('Courier New'); hoja.setFrozenRows(1); hoja.setColumnWidths(1, h.length, 120); } } function initSheet() { Object.values(HOJAS).forEach(n => getSheet(n)); const ss = getSS(); // Eliminar Sheet1 por defecto si existe const defaultSheet = ss.getSheetByName('Sheet1') || ss.getSheetByName('Hoja 1'); if (defaultSheet && ss.getSheets().length > 1) { ss.deleteSheet(defaultSheet); } return { ok: true, ssId: ss.getId(), ssUrl: ss.getUrl() }; } // ════════════════════════════════════════ // FICHAJES // ════════════════════════════════════════ function registrarEntrada(data) { // Solo registramos; la entrada se confirma al hacer salida return { ok: true, msg: 'Entrada recibida', timestamp: new Date().toISOString() }; } function registrarSalida(data) { const hoja = getSheet(HOJAS.FICHAJES); const { userId, nombre, fecha, entrada, salida, duracionMs, horas, horasExtra, pausas } = data; const jornadaH = horas - horasExtra; const id = 'F' + Date.now(); hoja.appendRow([ id, userId, nombre, fecha, entrada, salida, parseFloat(horas).toFixed(2), parseFloat(jornadaH).toFixed(2), parseFloat(horasExtra).toFixed(2), pausas || 0, new Date().toISOString() ]); return { ok: true, id }; } function getFichajes(params) { const hoja = getSheet(HOJAS.FICHAJES); const data = hoja.getDataRange().getValues(); if (data.length <= 1) return { ok: true, fichajes: [] }; const headers = data[0]; let rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); if (params.userId) rows = rows.filter(r => r.USER_ID === params.userId); if (params.mes !== undefined && params.anyo !== undefined) { rows = rows.filter(r => { const d = new Date(r.FECHA); return d.getMonth() === parseInt(params.mes) && d.getFullYear() === parseInt(params.anyo); }); } return { ok: true, fichajes: rows }; } // ════════════════════════════════════════ // VACACIONES // ════════════════════════════════════════ function guardarVacacion(data) { const hoja = getSheet(HOJAS.VACACIONES); const { id, userId, nombre, inicio, fin, motivo, estado, byAdmin } = data; // Comprobar si ya existe (update) const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === id) { hoja.getRange(i + 1, 1, 1, rows[0].length).setValues([[ id, userId, nombre||'', inicio, fin, motivo||'', estado||'pendiente', byAdmin||false, new Date().toISOString() ]]); return { ok: true, action: 'updated' }; } } hoja.appendRow([id, userId, nombre||'', inicio, fin, motivo||'', estado||'pendiente', byAdmin||false, new Date().toISOString()]); return { ok: true, action: 'created' }; } function actualizarVacacion(data) { const hoja = getSheet(HOJAS.VACACIONES); const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { hoja.getRange(i + 1, 7).setValue(data.estado); // col ESTADO hoja.getRange(i + 1, 9).setValue(new Date().toISOString()); return { ok: true }; } } return { ok: false, error: 'Vacación no encontrada' }; } function getVacaciones(params) { const hoja = getSheet(HOJAS.VACACIONES); const data = hoja.getDataRange().getValues(); if (data.length <= 1) return { ok: true, vacaciones: [] }; const headers = data[0]; let rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); if (params.userId) rows = rows.filter(r => r.USER_ID === params.userId); if (params.estado) rows = rows.filter(r => r.ESTADO === params.estado); return { ok: true, vacaciones: rows }; } // ════════════════════════════════════════ // WORKERS // ════════════════════════════════════════ function guardarWorker(data) { const hoja = getSheet(HOJAS.WORKERS); const { id, nombre, email, jornada, color, bg, role, activo, pin } = data; const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === id) { hoja.getRange(i + 1, 1, 1, 10).setValues([[ id, nombre, email||'', jornada||8, color||'', bg||'', role||'worker', activo!==false, pin||'1234', new Date().toISOString() ]]); return { ok: true, action: 'updated' }; } } hoja.appendRow([id, nombre, email||'', jornada||8, color||'', bg||'', role||'worker', activo!==false, pin||'1234', new Date().toISOString()]); return { ok: true, action: 'created' }; } function eliminarWorker(data) { const hoja = getSheet(HOJAS.WORKERS); const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { hoja.getRange(i + 1, 8).setValue(false); // activo = false return { ok: true }; } } return { ok: false, error: 'Worker no encontrado' }; } function getWorkers() { const hoja = getSheet(HOJAS.WORKERS); const data = hoja.getDataRange().getValues(); if (data.length <= 1) return { ok: true, workers: [] }; const headers = data[0]; const rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { ok: true, workers: rows }; } // ════════════════════════════════════════ // SYNC COMPLETO (pull inicial) // ════════════════════════════════════════ function sincronizarTodo(data) { // El cliente envía todo su localStorage → guardamos en Sheets if (data.fichajes) { data.fichajes.forEach(f => { const w = data.workers?.find(w => w.id === f.userId); registrarSalida({ ...f, nombre: w?.nombre || f.userId }); }); } if (data.vacaciones) { data.vacaciones.forEach(v => { const w = data.workers?.find(w => w.id === v.userId); guardarVacacion({ ...v, nombre: w?.nombre || v.userId }); }); } if (data.workers) { data.workers.forEach(w => guardarWorker(w)); } return { ok: true, synced: true }; } // ════════════════════════════════════════ // EXPORT PARA INSPECCIÓN // ════════════════════════════════════════ function getExport(params) { const mes = parseInt(params.mes); const anyo = parseInt(params.anyo); const uid = params.userId; const hoja = getSheet(HOJAS.FICHAJES); const data = hoja.getDataRange().getValues(); if (data.length <= 1) return { ok: true, rows: [] }; const headers = data[0]; let rows = data.slice(1) .map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) .filter(r => { const d = new Date(r.FECHA); const matchUser = !uid || uid === 'all' || r.USER_ID === uid; return matchUser && d.getMonth() === mes && d.getFullYear() === anyo; }) .map(r => ({ Empresa: PropertiesService.getScriptProperties().getProperty('EMPRESA') || 'Mi Empresa', Empleado: r.NOMBRE, DNI_NIE: '', Fecha: r.FECHA, 'Hora Entrada': r.ENTRADA, 'Hora Salida': r.SALIDA, 'Horas Totales': r.HORAS_TOTAL, 'Horas Ordinarias': r.HORAS_ORD, 'Horas Extra': r.HORAS_EXTRA, })); return { ok: true, rows }; } // ════════════════════════════════════════ // HELPER // ════════════════════════════════════════ function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function corsResponse(data) { // Para GET con JSONP si fuera necesario return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ════════════════════════════════════════ // TRIGGER: Reporte semanal por email (opcional) // ════════════════════════════════════════ function setupTriggers() { // Ejecutar una vez desde el editor para crear el trigger ScriptApp.newTrigger('reporteSemanal') .timeBased().onWeekDay(ScriptApp.WeekDay.MONDAY).atHour(8).create(); } function reporteSemanal() { const ss = getSS(); const hoja = getSheet(HOJAS.FICHAJES); const data = hoja.getDataRange().getValues(); if (data.length <= 1) return; const hoy = new Date(); const lunes = new Date(hoy); lunes.setDate(hoy.getDate() - 7); const viernesPas = new Date(hoy); viernesPas.setDate(hoy.getDate() - 3); const headers = data[0]; const semana = data.slice(1) .map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) .filter(r => { const d=new Date(r.FECHA); return d>=lunes && d<=viernesPas; }); if (!semana.length) return; const adminEmail = Session.getActiveUser().getEmail(); const tabla = semana.map(r => `${r.NOMBRE}${r.FECHA}${r.ENTRADA}${r.SALIDA}${r.HORAS_TOTAL}h${r.HORAS_EXTRA}h` ).join(''); const html = `

`; GmailApp.sendEmail(adminEmail, `FichaJA — Resumen semana ${lunes.toLocaleDateString('es-ES')}`, '', { htmlBody: html }); }
